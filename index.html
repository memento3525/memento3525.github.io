public enum UnitTeamID : byte
{
    None = 0,
    Player = 10,
    Enemy = 20,
}
 
/// <summary>
/// 유닛 팀을 나타내는 enum 객체입니다.
/// 각 팀 별로 달라지는 행동을 지정합니다.
/// </summary>
[DefaultExecutionOrder(-999)]
public class UnitTeam : IEquatable<UnitTeam>
{
    #region Static Instance
    public static readonly UnitTeam Player;
    public static readonly UnitTeam Enemy;
    public static readonly UnitTeam None;
 
    static UnitTeam()
    {
        Player = new UnitTeam(UnitTeamID.Player, 1000, 1999);
        Enemy = new UnitTeam(UnitTeamID.Enemy, 2000, 2999);
        None = new UnitTeam(UnitTeamID.None, 0, 999);
    }
 
    public static UnitTeam GetTeamInstance(UnitTeamID teamID)
    {
        return teamID switch
        {
            UnitTeamID.Player => Player,
            UnitTeamID.Enemy => Enemy,
            _ => None,
        };
    }
    #endregion
 
    #region Instance
    public readonly UnitTeamID teamID;
    public readonly Color teamColor;
    public readonly int layer;
    public readonly int layerMask;
 
    readonly UnitTeamID _opponentTeamID;
    readonly int _startUID;
    readonly int _maxUID;
    int _lastUID;
 
    public bool IsPlayerTeam() => teamID == UnitTeamID.Player;
    public bool IsEnemyTeam() => teamID == UnitTeamID.Enemy;
 
    public UnitTeam OpponentTeam => GetTeamInstance(_opponentTeamID);
 
    public UnitTeam(UnitTeamID teamID, int startUID, int maxUID)
    {
        _startUID = startUID;
        _maxUID = maxUID;
        _lastUID = startUID;
        this.teamID = teamID;
 
        _opponentTeamID = InitOpponentTeamID(teamID);
        teamColor = InitTeamColor(teamID);
        layer = InitLayer(teamID);
        layerMask = InitLayerMask(teamID);
    }
 
    UnitTeamID InitOpponentTeamID(UnitTeamID teamID)
    {
        return teamID switch
        {
            UnitTeamID.Player => UnitTeamID.Enemy,
            UnitTeamID.Enemy => UnitTeamID.Player,
            _ => UnitTeamID.None,
        };
    }
 
    Color InitTeamColor(UnitTeamID teamID)
    {
        return teamID switch
        {
            UnitTeamID.Player => new Color(0f, 0.7f, 0.8f),
            UnitTeamID.Enemy => new Color(0.8f, 0.3f, 0f),
            _ => Color.white,
        };
    }
 
    int InitLayer(UnitTeamID teamID)
    {
        return teamID switch
        {
            UnitTeamID.Player => MentumLayer.Player,
            UnitTeamID.Enemy => MentumLayer.Enemy,
            _ => MentumLayer.Player,
        };
    }
 
    int InitLayerMask(UnitTeamID teamID)
    {
        return teamID switch
        {
            UnitTeamID.Player => MentumLayerMask.Player,
            UnitTeamID.Enemy => MentumLayerMask.Enemy,
            _ => MentumLayerMask.Player,
        };
    }
 
    public override string ToString()
    {
        return teamID switch
        {
            UnitTeamID.Player => "Player",
            UnitTeamID.Enemy => "Enemy",
            _ => "None",
        };
    }
    #endregion
 
    #region UID
    public int NewUID()
    {
        _lastUID++;
 
        if (_lastUID > _maxUID)
            _lastUID = _startUID;
 
        return _lastUID;
    }
 
    public string GetPrefix()
    {
        return teamID switch
        {
            UnitTeamID.Player => "P",
            UnitTeamID.Enemy => "E",
            _ => "N",
        };
    }
    #endregion
 
    #region IEquatable
    public static bool operator ==(UnitTeam a, UnitTeam b)
    {
        return a.teamID == b.teamID;
    }
 
    public static bool operator !=(UnitTeam a, UnitTeam b)
    {
        return a.teamID != b.teamID;
    }
 
    public override bool Equals(object obj)
    {
        if (obj is not UnitTeam otherTeam)
            return false;
 
        return teamID == otherTeam.teamID;
    }
 
    public bool Equals(UnitTeam other)
    {
        if (other == null)
            return false;
 
        return teamID == other.teamID;
    }
 
    public override int GetHashCode()
    {
        return this.GetHashCode();
    }
    #endregion
}

/// <summary>
/// 조각코드1 : 같은 배틀섹션에 속하지 않은 새로운 상대 그룹이 탐지되면 반환한다.
/// </summary>
public bool TryDetectEnemyGroups(BattleUnitGroup ownerGroup, float sqrDetectRadius)
{
    if (!TryGetTeamContainer(ownerGroup.UnitTeam.OpponentTeam, out var opponentTeamContainer))
        return false;
    bool anyGroupDetected = false;
    bool ownerGroupHasBattleSection = ownerGroup.BattleSection != null;
    foreach (var enemyGroup in opponentTeamContainer.UnitGroupIEnumerable)
    {
        if (!enemyGroup.IsActiveGroup) 
            continue;
        if (ownerGroupHasBattleSection && ownerGroup.BattleSection == enemyGroup.BattleSection) 
            continue;
        float sqrDistance = (ownerGroup.GroupCenterPosition - enemyGroup.GroupCenterPosition).sqrMagnitude;
        if (sqrDistance <= sqrDetectRadius)
        {
            // Debug.Log($"{ownerGroup.BattleSection} 에서 {ownerGroup}가 {detectGroup}를 발견하여 매칭함");
            BattleSection.MatchWith(ownerGroup, enemyGroup); 
            anyGroupDetected = true;
            // 복수의 그룹이 매칭될 수 있어 return 하진 않습니다.
        }
    }
    return anyGroupDetected;
}
/// <summary>
/// 조각코드2 : 유닛 레벨업시 변경점 받는 콜백함수
/// </summary>
void OnUnitLevelUp(UnitLevelUp_t e)
{
    if (e.level == null || e.unitDataKey == null) return;
    if (!TryGetTeamContainer(UnitTeam.Player, out var teamContainer)) return;
    foreach (var group in teamContainer.UnitGroupIEnumerable)
    {
        if (group.Leader == null) continue;
        if (!group.Leader.Key.Equals(e.unitDataKey)) continue;
        foreach (var unit in group.StartingUnits)
        {
            if (unit != null)
                unit.UpdateLevel(e.level);
        }
    }
}
